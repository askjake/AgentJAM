📝 USAGE EXAMPLES:
══════════════════

# Simple query (auto-routes to Haiku/Sonnet)
python -m agentjam.main "Is the API healthy?"

# Standard analysis (uses Sonnet)
python -m agentjam.main "Analyze this error log"

# Complex reasoning (forces Opus)
python -m agentjam.main --reasoning "Design disaster recovery strategy"

# Fast mode (forces Haiku)
python -m agentjam.main --fast "Check pod status"

# Explicit model
python -m agentjam.main --model opus "Complex query"

# JSON output
python -m agentjam.main --json "Your query" > output.json

# Verbose logging
python -m agentjam.main --verbose "See model selection reasoning"
